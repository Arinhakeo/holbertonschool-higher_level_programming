# Test cases for 1-my_list.py

# Importing the MyList class
>>> MyList = __import__('1-my_list').MyList

# Test case 1: Creating an empty list and printing it sorted
>>> my_list = MyList()
>>> my_list.print_sorted()
[]

# Test case 2: Adding elements and printing sorted
>>> my_list.append(1)
>>> my_list.append(4)
>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.append(5)
>>> print(my_list)
[1, 4, 2, 3, 5]
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]
>>> print(my_list)
[1, 4, 2, 3, 5]

# Test case 3: List with negative numbers
>>> my_list = MyList()
>>> my_list.append(-1)
>>> my_list.append(-4)
>>> my_list.append(-2)
>>> my_list.append(-3)
>>> my_list.append(-5)
>>> my_list.print_sorted()
[-5, -4, -3, -2, -1]

# Test case 4: List with duplicate numbers
>>> my_list = MyList()
>>> my_list.append(1)
>>> my_list.append(1)
>>> my_list.append(2)
>>> my_list.append(2)
>>> my_list.append(1)
>>> my_list.print_sorted()
[1, 1, 1, 2, 2]

# Test case 5: List with a single element
>>> my_list = MyList()
>>> my_list.append(42)
>>> my_list.print_sorted()
[42]

# Test case 6: List with zero
>>> my_list = MyList()
>>> my_list.append(0)
>>> my_list.append(-1)
>>> my_list.append(1)
>>> my_list.append(0)
>>> my_list.print_sorted()
[-1, 0, 0, 1]

# Test case 7: Ensure original list is not modified
>>> my_list = MyList([3, 1, 2])
>>> my_list.print_sorted()
[1, 2, 3]
>>> print(my_list)
[3, 1, 2]
